let allData = []; const fmt = new Intl.NumberFormat('id-ID'); document.getElementById('fetchBtn').onclick = async function() { const sId = document.getElementById('storeId').value.trim().toUpperCase(); const rk = document.getElementById('filterRak').value.trim().toUpperCase(); if(!sId) return alert("Isi Kode Toko!"); this.textContent = 'Memuat...'; this.disabled = true; try { const response = await fetch(`/proxy-products?storeId=${sId}&rack=${rk}`); const result = await response.json(); // LOGIKA DETEKSI DATA: // Kadang API balikin [ {...} ], kadang { "data": [...] } if (Array.isArray(result)) { allData = result; } else if (result.data && Array.isArray(result.data)) { allData = result.data; } else if (result.product_list && Array.isArray(result.product_list)) { allData = result.product_list; } else { allData = []; } if(allData.length > 0) { document.getElementById('statsEl').classList.remove('hidden'); document.getElementById('tableCard').classList.remove('hidden'); document.getElementById('s-store').textContent = sId; document.getElementById('s-total').textContent = allData.length; renderTable(); } else { alert(`Data tidak ditemukan! Pastikan Kode Toko (${sId}) dan Rak (${rk}) sudah benar.`); } } catch (e) { alert("Error: " + e.message); } finally { this.textContent = 'Tampilkan'; this.disabled = false; } }; function renderTable() { const q = document.getElementById('search').value.toLowerCase(); const tbody = document.getElementById('tbody'); const filtered = allData.filter(i => (i.descp || "").toLowerCase().includes(q) || (i.plu || "").toString().includes(q) ); tbody.innerHTML = filtered.map(item => { // Ambil harga dari berbagai kemungkinan nama field const harga = item.price || item.price_sale || item.salesPrice || 0; return ` ${Math.round(item.plu || 0)} ${item.descp || '-'} ${item.planogram || item.rack || rk || '-'} ${item.barcode || '-'} ${fmt.format(harga)} ${item.subdept || '-'} `; }).join(''); }
